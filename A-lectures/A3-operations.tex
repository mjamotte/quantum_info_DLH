\begin{frame}{Opérateurs quantiques sur un qubit}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.55em}
    \item Les \textbf{vecteurs} représentant les \textbf{qubits} sont transformés par des \textbf{matrices}
     représentant les \textbf{opérateurs quantiques}.\\~\\
    \item Exemples d'opérateurs à 1 qubit: Pauli X (inversion), Z (phase), Hadamard $H$ (création de superposition).\\~\\
  \end{itemize}
\end{frame}

\begin{frame}{Opération $Z$ (phase)}
  Notation matricielle:
  $$
    Z = \begin{pmatrix}1 & 0 \\ 0 & -1\end{pmatrix}
  $$\\~\\
  $$
    \begin{pmatrix}1 & 0 \\ 0 & -1\end{pmatrix}
    \begin{pmatrix}1\\0\end{pmatrix}=
    \begin{pmatrix}1\\0\end{pmatrix},
    \qquad
    \begin{pmatrix}1 & 0 \\ 0 & -1\end{pmatrix}
    \begin{pmatrix}0\\1\end{pmatrix}=
    \begin{pmatrix}0\\-1\end{pmatrix}
  $$
  ~\\~\\
  Notation d'opérateur: $$Z = \ket 0 \bra 0 - \ket 1 \bra 1$$\\~\\
  $$
    Z\ket 0 = \ket 0, \qquad
    Z\ket 1 = -\ket 1
  $$
  
\end{frame}


\begin{frame}{Opération $X$ (NOT quantique)}
  Notation matricielle:
  $$
    X = \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}
  $$\\~\\
  $$
    \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}
    \begin{pmatrix}1\\0\end{pmatrix}=
    \begin{pmatrix}0\\1\end{pmatrix},
    \qquad
    \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}
    \begin{pmatrix}0\\1\end{pmatrix}=
    \begin{pmatrix}1\\0\end{pmatrix}
  $$
  ~\\~\\
  Notation d'opérateur: $$X = \ket 0 \bra 1 + \ket 1 \bra 0$$\\~\\
  $$
    X\ket 0 = \ket 1, \qquad
    X\ket 1 = \ket 0
  $$
  
\end{frame}



\begin{frame}{Opération $H$ (Hadamard)}
  Notation matricielle:
  $$
    H = \tfrac{1}{\sqrt{2}}\begin{pmatrix}1 & 1 \\ 1 & -1\end{pmatrix}
  $$\\~\\
  $$
    \tfrac{1}{\sqrt{2}}\begin{pmatrix}1 & 1 \\ 1 & -1\end{pmatrix}
    \begin{pmatrix}1\\0\end{pmatrix}=
    \tfrac{1}{\sqrt{2}}\begin{pmatrix}1\\1\end{pmatrix},
    \qquad
    \tfrac{1}{\sqrt{2}}\begin{pmatrix}1 & 1 \\ 1 & -1\end{pmatrix}
    \begin{pmatrix}0\\1\end{pmatrix}=
    \tfrac{1}{\sqrt{2}}\begin{pmatrix}1\\-1\end{pmatrix}
  $$
  ~\\~\\
  Notation d'opérateur: $$H = \tfrac{1}{\sqrt{2}} \ket 0 \bra 0 + \tfrac{1}{\sqrt{2}} \ket 0 \bra 1 + \tfrac{1}{\sqrt{2}} \ket 1 \bra 0 - \tfrac{1}{\sqrt{2}} \ket 1 \bra 1$$\\~\\
  $$
    H\ket 0 = \tfrac{1}{\sqrt{2}}(\ket 0+\ket 1), \qquad
    H\ket 1 = \tfrac{1}{\sqrt{2}}(\ket 0-\ket 1)
  $$
\end{frame}
