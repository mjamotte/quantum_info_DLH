\begin{frame}{Truth table}
  \centering
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{l c c c}
    \textbf{Gate} & \textbf{Symbol} & \textbf{Expression} & \textbf{Description} \\\hline
    NOT   & $\lnot A$ & inverse & $1\leftrightarrow 0$ \\
    AND   & $A\cdot B$ & $1$ if $A=1$ and $B=1$ & "multiplication" gate \\
    OR    & $A + B$ & $1$ if $A=1$ or $B=1$ & "addition" gate \\
    XOR   & $A \oplus B$ & $1$ if $A\neq B$ & "exclusive or" \\
    NAND  & $\lnot(A\cdot B)$ & inverse of AND & universal basis \\
  \end{tabular}
\end{frame}

\begin{frame}{Running a circuit on a QPU}
  \begin{itemize}
    \item Circuit creation\\~\\
    \item Choice of execution platform (cloud or local)
    \begin{enumerate}
      \vspace{0.5em}
    \item `QPU`: executes the circuit on a real IBM quantum processor via Qiskit Runtime, with hardware noise;
    \item `FakeQPU`: mimics an IBM QPU from a snapshot (topology, native gates, qubit properties) to test transpilation and noisy simulation;
    \item `Aer`: simulates the circuit locally with `AerSimulator`, quickly and with several simulation methods, with or without a noise model
    \end{enumerate}~\\
    \item Transpilation of the circuit to adapt it to the QPU\\~\\
  \end{itemize}
\end{frame}

\begin{frame}{Transpilation: Six Steps}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \small
      \vspace{-0.7em}
      \begin{enumerate}
        \setlength{\itemsep}{0.3em}
        \item \textbf{Init}: Decomposition of multi-qubit operations into 1- and 2-qubit gates (Optional)
        \item \textbf{Layout}: Assignment of physical qubits
        \item \textbf{Routing}: Insertion of SWAP gates to move quantum states between qubits (NP-hard, randomized algorithm)
        \item \textbf{Translation}: Conversion of all gates to the native instruction set (ISA)
        \item \textbf{Optimization}: Replacement or elimination of logic gates where possible
        \item \textbf{Scheduling}: Insertion of delays (Optional)
      \end{enumerate}
    \end{column}
    \begin{column}{0.4\textwidth}
      \centering
      \includegraphics[width=\textwidth]{../figures/qubits_layout.jpg}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Transpilation: Optimization Levels}
  \small
  \begin{description}
    \setlength{\itemsep}{1.5em}
    \item[\textbf{Level 1}:] Matrix combination of single-qubit gates,\\ elimination of redundant CNOT gates
    
    \item[\textbf{Level 2}:] Commutation analysis to eliminate further redundancies
    
    \item[\textbf{Level 3}:] Optimization of two-qubit sub-blocks identified in the circuit\\ (reconstructs an equivalent unitary matrix)
  \end{description}
\end{frame}
