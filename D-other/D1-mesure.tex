\begin{frame}{Measuring a qubit}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.6em}
    \item General state of a qubit: $\ket{\psi} = \alpha \ket 0 + \beta \ket 1$.
    \\~\\
    \item Measuring a qubit gives \textbf{a single result}: $\ket{0}$ or $\ket{1}$ (in the computational basis).
    \\~\\
    \item The probabilities are given by the \textbf{squared coefficients}:
    $$P(\ket 0) = |\alpha|^2, \qquad P(\ket 1) = |\beta|^2$$
    \item After measurement, the state is \textbf{projected} onto the obtained result ($\ket{0}$ or $\ket{1}$).
  \end{itemize}
\end{frame}

\begin{frame}{Projection in quantum mechanics: ket-bra notation}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.6em}
    \item The state of a qubit in the form $\ket \psi$ is written $\begin{pmatrix}
    \alpha\\ \beta \end{pmatrix}$ \\~\\
    \item The state of a qubit in the form $\bra \psi$ is written $\begin{pmatrix}
    \alpha \ \beta \end{pmatrix}$ (if $\alpha, \beta \in \mathbb R$) \\~\\
    \item Projector onto the state $\ket \psi$: $$\Pi_\psi^{} = \ket \psi \bra \psi$$ 
  \end{itemize}
\end{frame}

\begin{frame}{Projectors in matrix notation}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.6em}
    \item $\Pi_\psi^{} = \ket \psi \bra \psi = (\alpha \  \beta) \begin{pmatrix}
    \alpha \\ \beta
    \end{pmatrix} = \begin{pmatrix}
      \alpha^2 & \alpha \beta\\
      \alpha \beta & \beta^2
    \end{pmatrix}$\\~\\
    \item The projectors $\Pi_0$ and $\Pi_1$ are also written as $2\times 2$ matrices:
    $$\Pi_0 = \ket{0}\bra{0} =
      \begin{pmatrix}1\\0\end{pmatrix}\begin{pmatrix}1 & 0\end{pmatrix}
      = \begin{pmatrix}1 & 0\\0 & 0\end{pmatrix}$$
    $$\Pi_1 = \ket{1}\bra{1} =
      \begin{pmatrix}0\\1\end{pmatrix}\begin{pmatrix}0 & 1\end{pmatrix}
      = \begin{pmatrix}0 & 0\\0 & 1\end{pmatrix}$$
    %\item On vérifie la relation de complétude:    $\Pi_0 + \Pi_1 = \begin{pmatrix}1 & 0\\0 & 1\end{pmatrix} = I$ (matrice identité).
    %\item La probabilité de mesurer $\ket{0}$ s'écrit: $P(0) = \bra{\psi}\Pi_0\ket{\psi} = |\alpha|^2$.
  \end{itemize}
\end{frame}

\begin{frame}{Projection, dot product and measurement probability}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.6em}
    \item Measuring $\ket \psi$ projects it into the state $\ket 0$ or $\ket 1$: 
    $$\Pi_0^{} \ket \psi = \alpha \ket 0, \quad \Pi_1^{} \ket \psi = \beta \ket 1$$
    \item The probability of measuring $\ket{0}$ or $\ket 1$ is written as the dot product of $\ket \psi$ with its projection onto the state $\ket 0$ or $\ket 1$:
    $$P(0) = \bra{\psi}\Pi_0\ket{\psi} = |\alpha|^2$$
    $$P(1) = \bra{\psi}\Pi_1\ket{\psi} = |\beta|^2$$
  \end{itemize}
\end{frame}

\section{Reproducibility of measurement}
% ============================================================

\begin{frame}{Reproducibility of measurement: $\Pi^2 = \Pi$}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.8em}
    \item Measurement \textbf{projects} the quantum state onto $\ket{0}$ or $\ket{1}$
     according to the probabilities $P(\ket 0)$ and $P(\ket 1)$.\\~\\
    \item If we measure again, we find the same result: the measurement is reproducible.\\~\\
    \item Mathematically: $\Pi^2 = \Pi$, which is why measurement is a projection.\\~\\
    \item The state $\ket 0$ remains unchanged under the action of $\Pi_0^{}$: $\Pi_0^{} \ket 0 = \ket 0$\\~\\ 
    \item $\ket 0$ is an \textbf{eigenstate} of the projector $\Pi_0^{}$.
  \end{itemize}
\end{frame}

\begin{frame}{Eigenvalues and eigenvectors}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.6em}
    \item We also say that the vector representing the state $\ket 0$ is an eigenvector of the matrix representing the projector $\Pi_0^{}$\\~\\
    \item Another example of a matrix-vector product:   
    $$ \begin{pmatrix} 2 & 1 \\ 2 & 3 \end{pmatrix} \begin{pmatrix} -1 \\ 1 \end{pmatrix} = \begin{pmatrix} -1 \\ 1 \end{pmatrix},$$
    and 
    $$ \begin{pmatrix} 2 & 1 \\ 2 & 3 \end{pmatrix} \begin{pmatrix} 1 \\ 2 \end{pmatrix} = \begin{pmatrix} 4 \\ 8 \end{pmatrix} = 4 \begin{pmatrix} 1 \\ 2 \end{pmatrix},$$~\\
    \item Multiplying this vector by this matrix gives back the same vector multiplied by a number.
    \item This vector is called an "\textbf{eigenvector}" and this number an "\textbf{eigenvalue}".
  \end{itemize}
\end{frame}

\begin{frame}{The states $\ket 0, \ket 1$ are eigenvectors of an observable}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.6em}
    \item Measurable quantities are represented by matrices called \textit{observables}.
    \item \textbf{Measuring} a physical quantity amounts to measuring one of its \textbf{eigenvalues}.
    \item The states $\ket{0}$ and $\ket{1}$ are the eigenvectors of the observable $Z$
     (computational basis):
    $$Z\ket{0} = +\ket{0}, \qquad Z\ket{1} = -\ket{1}$$
    \item The physical nature of the eigenvalue (energy, spin,...) depends on the platform (see later).
    \item \textbf{Postulate}: measuring = projecting the qubit vector onto the eigenvector associated with the
     eigenvalue of the measured observable.
  \end{itemize}
\end{frame}

\section{Other bases and observables}
% ============================================================

\begin{frame}{Projection onto arbitrary vectors}
  \begin{itemize}[<+->]
    \setlength{\itemsep}{0.8em}
    \item One can measure in bases other than the computational basis $\{\ket{0}, \ket{1}\}$.\\~\\
    \item For example, the basis $\{\ket{+}, \ket{-}\}$ with:
    $$\ket{+} = \frac{\ket{0}+\ket{1}}{\sqrt{2}}, \qquad \ket{-} = \frac{\ket{0}-\ket{1}}{\sqrt{2}}$$
    \item The vectors $\ket{+}$ and $\ket{-}$ are the eigenvectors of the observable $X$:
    $$X\ket{+} = (+1)\ket{+}, \qquad X\ket{-} = (-1)\ket{-}$$
    \item One can thus define projectors and measurement probabilities in this basis.
  \end{itemize}
\end{frame}
